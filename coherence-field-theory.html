    "    \"\"\"Evolve symbolic coherence field dynamics\"\"\"\\n",
    "    nx, ny = theta.shape\\n",
    "    theta_new = theta.copy()\\n",
    "    \\n",
    "    for i in range(1, nx-1):\\n",
    "        for j in range(1, ny-1):\\n",
    "            # Gradients\\n",
    "            dtheta_dx = (theta[i+1,j] - theta[i-1,j]) / (2*dx)\\n",
    "            dtheta_dy = (theta[i,j+1] - theta[i,j-1]) / (2*dx)\\n",
    "            \\n",
    "            # Laplacian\\n",
    "            laplacian = (theta[i+1,j] + theta[i-1,j] + theta[i,j+1] + theta[i,j-1] - 4*theta[i,j]) / (dx*dx)\\n",
    "            \\n",
    "            # Convective term\\n",
    "            convective = dtheta_dx * dtheta_dx + dtheta_dy * dtheta_dy\\n",

    "            # Coherence tension\\n",
